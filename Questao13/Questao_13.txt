void entrada(int tam, float num[]){
  printf("Digite os numeros para serem ordenados: \n");
  for(int i=0; i<tam; i++){

  }
}

void ordem(int tam, float num[]){
  for( int i=0; i<tam; i++ ){
    for( int j=i+1; j<tam; j++ ){  
      if( num[i] > num[j] ){
        float aux;
        aux =num[i];
        num[i] = num[j];
        num[j] = aux;
      }
    }
  }
}
