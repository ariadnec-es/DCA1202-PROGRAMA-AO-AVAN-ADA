#include <stdio.h>
#include <stdlib.h>

void ordem(float *v, int tam){
    for(int i=0; i<tam-1; i++){
        for(int j=i; j<tam; j++){
            if(v[i]>v[j]){
                int aux = v[i];
                v[i] = v[j];
                v[j] = aux;
            }
        }
    }
}

int main(){
    float *p; // Ponteiro para o vetor
    int i, qtdElementos;

    printf("Digite a qtd de elementos do vetor: ");
    scanf("%d", &qtdElementos);

    //Alocando a memória para o vetor de acordo com a quantidade de elementos do vetor
    p = (float*)(malloc(qtdElementos * sizeof(float)));

    //teste de alocação
    if(p==NULL){
        printf("ERRO");
        system("Pause");
        exit(1);
    }

    printf("\n");

    //recebe dados
    printf("Digite os numeros para ordenar:\n");
    for(int i=0;i<qtdElementos;i++){
        scanf("%f", &p[i]);
    }

    //Função de Ordenação Crescente
    ordem(p,qtdElementos);

    printf("\n");

    printf("------------ Ordenado de forma crescente ------------\n");

    //Mostra
    for(int i = 0; i < qtdElementos; i++)
        printf("%.1f ", p[i]);


}
